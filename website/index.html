        <div id="score-display">Score: 0</div>

        <div id="time-display">Time: 60s</div>

        

        <div id="start-screen">

            <h1>Bubble Pop!</h1>

            <p>Pop as many bubbles as you can before time runs out!</p>

            <p>Larger bubbles = more points</p>

            <button id="start-button">Start Game</button>

        </div>

        

        <div id="game-over-screen">

            <h1>Game Over!</h1>

            <div id="final-score">Your Score: 0</div>

            <button id="restart-button">Play Again</button>

        </div>

    </div>


    <script>

        document.addEventListener('DOMContentLoaded', function() {

            // Game variables

            let score = 0;

            let timeLeft = 60;

            let gameRunning = false;

            let gameTimer;

            let bubbleSpawnRate = 1000; // ms between bubble spawns

            let spawnTimer;

            

            // DOM elements

            const gameContainer = document.getElementById('game-container');

            const scoreDisplay = document.getElementById('score-display');

            const timeDisplay = document.getElementById('time-display');

            const startScreen = document.getElementById('start-screen');

            const gameOverScreen = document.getElementById('game-over-screen');

            const finalScore = document.getElementById('final-score');

            const startButton = document.getElementById('start-button');

            const restartButton = document.getElementById('restart-button');

            

            // Bubble colors

            const bubbleColors = [

                '#FF5733', '#33FF57', '#3357FF', '#FF33F5', 

                '#F5FF33', '#33FFF5', '#FF3333', '#33FF33'

            ];

            

            // Event listeners

            startButton.addEventListener('click', startGame);

            restartButton.addEventListener('click', startGame);

            

            // Start the game

            function startGame() {

                // Reset game state

                score = 0;

                timeLeft = 60;

                scoreDisplay.textContent = 'Score: 0';

                timeDisplay.textContent = 'Time: 60s';

                

                // Clear any existing bubbles

                const bubbles = document.querySelectorAll('.bubble');

                bubbles.forEach(bubble => bubble.remove());

                

                // Hide start/game over screens

                startScreen.style.display = 'none';

                gameOverScreen.style.display = 'none';

                

                // Start game

                gameRunning = true;

                

                // Start timers

                gameTimer = setInterval(updateTime, 1000);

                spawnTimer = setInterval(createBubble, bubbleSpawnRate);

                

                // Create initial bubbles

                for (let i = 0; i < 10; i++) {

                    createBubble();

                }

            }

            

            // Update timer

            function updateTime() {

                timeLeft--;

                timeDisplay.textContent = `Time: ${timeLeft}s`;

                

                if (timeLeft <= 0) {

                    endGame();

                }

                

                // Increase spawn rate as time progresses

                if (timeLeft % 10 === 0 && bubbleSpawnRate > 200) {

                    bubbleSpawnRate -= 100;

                    clearInterval(spawnTimer);

                    spawnTimer = setInterval(createBubble, bubbleSpawnRate);

                }

            }

            

            // Create a bubble

            function createBubble() {

                if (!gameRunning) return;

                

                const bubble = document.createElement('div');

                bubble.className = 'bubble';

                

                // Random size (larger bubbles worth more points)

                const size = Math.random() * 80 + 40; // 40-120px

                bubble.style.width = `${size}px`;

                bubble.style.height = `${size}px`;

                

                // Calculate points based on size (smaller bubbles worth more)

                const points = Math.floor((140 - size) / 10);

                bubble.dataset.points = points;

                bubble.textContent = points;

                

                // Random position

                const maxX = window.innerWidth - size;

                const maxY = window.innerHeight - size;

                bubble.style.left = `${Math.random() * maxX}px`;

                bubble.style.top = `${Math.random() * maxY}px`;

                

                // Random color

                const colorIndex = Math.floor(Math.random() * bubbleColors.length);

                bubble.style.backgroundColor = bubbleColors[colorIndex];

                

                // Random animation delay for floating effect

                bubble.style.animationDelay = `${Math.random() * 2}s`;

                

                // Add click event

                bubble.addEventListener('click', popBubble);

                

                // Add to game

                gameContainer.appendChild(bubble);

                

                // Auto-remove bubble after some time to keep screen from getting too cluttered

                setTimeout(() => {

                    if (bubble.parentNode === gameContainer) {

                        bubble.remove();

                    }

                }, 8000);

            }

            

            // Pop a bubble

            function popBubble(e) {

                if (!gameRunning) return;

                

                const bubble = e.target;

                const points = parseInt(bubble.dataset.points);

                

                // Add points to score

                score += points;

                scoreDisplay.textContent = `Score: ${score}`;

                

                // Create score popup

                const scorePopup = document.createElement('div');

                scorePopup.className = 'score-popup';

                scorePopup.textContent = `+${points}`;

                scorePopup.style.left = `${bubble.offsetLeft + parseInt(bubble.style.width) / 2}px`;

                scorePopup.style.top = `${bubble.offsetTop}px`;

                scorePopup.style.color = bubble.style.backgroundColor;

                gameContainer.appendChild(scorePopup);

                setTimeout(() => scorePopup.remove(), 1000);

                

                // Play pop animation and remove

                bubble.classList.add('pop-animation');

                setTimeout(() => bubble.remove(), 300);

                

                // Create a new bubble to replace the popped one

                createBubble();

            }

            

            // End the game

            function endGame() {

                gameRunning = false;

                clearInterval(gameTimer);

                clearInterval(spawnTimer);

                

                finalScore.textContent = `Your Score: ${score}`;

                gameOverScreen.style.display = 'flex';

            }

            

            // Resize handler

            window.addEventListener('resize', function() {

                // Adjust bubble positions if they're outside the viewport

                const bubbles = document.querySelectorAll('.bubble');

                bubbles.forEach(bubble => {

                    const rect = bubble.getBoundingClientRect();

                    const size = parseInt(bubble.style.width);

                    

                    if (rect.right > window.innerWidth) {

                        bubble.style.left = `${window.innerWidth - size}px`;

                    }

                    

                    if (rect.bottom > window.innerHeight) {

                        bubble.style.top = `${window.innerHeight - size}px`;

                    }

                });

            });

        });

    </script>
